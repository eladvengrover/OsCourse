int handle_output_op(int count, char** arglist) {
    int fd = open(arglist[count - 1], O_WRONLY);
    if (fd == -1) {
        return 0;
    }
    int pid = fork();
    if (pid == -1) {
        return -1;
    }
    if (pid == 0) {
        // Child proccess
        if (dup2(fd, STDOUT_FILENO) == -1) {
            return -1;
        }
        close(fd);
        if (execvp(arglist[0], arglist) == -1) {
            perror("execvp failed!");
            exit(1);
        }
    } else {
        // Parent proccess
        if (waitpid(pid, NULL, 0) == -1) {
            return 0;
        }
        return 1;
    }
}
